Hello World

akashic initコマンドで出力されるテンプレートです。
画面上部を赤い矩形が左から右に移動する動作を繰り返します。

  • game.json
  • main.js
{
	"width": 320,
	"height": 320,
	"fps": 30,
	"main": "./script/main.js",
	"assets": {
		"main": {
			"type": "script",
			"path": "script/main.js",
			"global": true
		}
	}
}
function main(param) {
	const scene = new g.Scene({game: g.game});
	scene.loaded.handle(function() {
		// 以下にゲームのロジックを記述します。
		const rect = new g.FilledRect({
			scene: scene,
			cssColor: "#ff0000",
			width: 32,
			height: 32
		});
		rect.update.handle(function () {
			// 以下のコードは毎フレーム実行されます。
			rect.x++;
			if (rect.x > g.game.width) rect.x = 0;
			rect.modified();
		});
		scene.append(rect);
	});
	g.game.pushScene(scene);
}

module.exports = main;
